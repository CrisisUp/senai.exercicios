   42 // --- 2. MODELO DE NEGÓCIO: CLASSE PRODUTO ---
   43 
   44 class Produto {
   45 private:
   46     string nome;
   47     int atual;
   48     int minimo;
   49     double preco;
   50 
   51 public:
   52     Produto(string n, int a, int m, double p) 
   53         : nome(n), atual(a), minimo(m), preco(p) {}
   54 
   55     // Getters e Lógica de Negócio (Encapsulamento)
   56     string getNome() const { return nome; }
   57     bool precisaRepor() const { return atual < minimo; }
   58     int getFalta() const { return (precisaRepor()) ? (minimo - atual) : 0; }
   59     double getCustoReposicao() const { return getFalta() * preco; }
   60     
   61     void exibirStatus() const {
   62         cout << UI::NEGRITO << left << setw(15) << nome << UI::RESET;
   63         cout << " | " << setw(5) << atual << " / " << setw(5) << minimo;
   64         if (precisaRepor()) {
   65             cout << UI::VERMELHO << " | FALTA: " << setw(4) << getFalta() 
   66                  << " | CUSTO: R$ " << fixed << setprecision(2) << getCustoReposicao() << UI::RESET << endl;
   67         } else {
   68             cout << UI::VERDE << " | ESTOQUE SUFICIENTE" << UI::RESET << endl;
   69         }
   70     }
   71 };
   72 
   73 // --- 3. LABORATORIO DE TESTES (VALIDAÇÃO ATIVA) ---
   74 
   75 class SuiteDeTestes {
   76 public:
   77     static void executar() {
   78         int falhas = 0;
   79         cout << UI::AMARELO << "\n--- TESTANDO LOGICA DE REPOSICAO ---" << UI::RESET << endl;
   80 
   81         // Caso 1: Estoque abaixo do mínimo
   82         Produto p1("Teste1", 5, 10, 2.0);
   83         if (p1.getFalta() != 5 || p1.getCustoReposicao() != 10.0) {
   84             cout << UI::VERMELHO << "[FALHA]: Cálculo de reposição incorreto para p1." << UI::RESET << endl;
   85             falhas++;
   86         } else {
   87             cout << UI::VERDE << "[OK]: Cálculo de falta e custo validado." << UI::RESET << endl;
   88         }
   89 
   90         // Caso 2: Estoque exatamente no mínimo
   91         Produto p2("Teste2", 10, 10, 2.0);
   92         if (p2.precisaRepor()) {
   93             cout << UI::VERMELHO << "[FALHA]: Produto no limite não deve repor." << UI::RESET << endl;
   94             falhas++;
   95         } else {
   96             cout << UI::VERDE << "[OK]: Limite de estoque validado." << UI::RESET << endl;
   97         }
   98 
   99         cout << UI::AMARELO << "-----------------------------------------------" << endl;
  100         if (falhas == 0) 
  101             cout << UI::VERDE << "RESULTADO: MOTOR DE ESTOQUE 100% CONFIAVEL.\n" << UI::RESET << endl;
  102         else
  103             exit(1); // Aborta se a lógica estiver errada (Rigor de Engenharia)
  104     }
  105 };
  106 
  107 // --- 4. FUNÇÃO PRINCIPAL ---
  108 
  109 int main()
  110 {
  111     UI::cabecalho();
  112 
  113     // EXIGÊNCIA DE ENGENHARIA: Validar antes de rodar
  114     SuiteDeTestes::executar();
  115 
  116     // Gestão Moderna com Smart Pointers e Coleções STL
  117     vector<unique_ptr<Produto>> estoque;
  118 
  119     // Cadastro de produtos (Simulando uma carga de banco de dados/arquivo)
  120     estoque.push_back(make_unique<Produto>("Monitor LED", 12, 15, 450.00));
  121     estoque.push_back(make_unique<Produto>("Teclado Mec", 45, 20, 180.00));
  122     estoque.push_back(make_unique<Produto>("Mouse Laser", 8, 25, 90.00));
  123 
  124     cout << UI::NEGRITO << left << setw(15) << "PRODUTO" << " | ESTOQUE / MIN | STATUS" << UI::RESET << endl;
  125     cout << "-----------------------------------------------" << endl;
  126 
  127     double custoTotalGeral = 0;
  128 
  129     for (const auto& item : estoque) {
  130         item->exibirStatus();
  131         custoTotalGeral += item->getCustoReposicao();
  132     }
  133 
  134     cout << "-----------------------------------------------" << endl;
  135     cout << UI::CIANO << "INVESTIMENTO TOTAL NECESSARIO: R$ " 
  136          << fixed << setprecision(2) << custoTotalGeral << UI::RESET << endl;
  137     cout << "===============================================" << endl;
  138 
  139     // Nota: Nenhum 'delete' manual é necessário. 
  140     // O unique_ptr limpa a memória automaticamente ao sair do escopo.
  141 
  142     return 0;
  143 }
