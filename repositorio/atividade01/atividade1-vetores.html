1 /**
    2  * @file atividade1-vetores.cpp
    3  * @brief Sistema de Análise e Controle de Estoque.
    4  * 
    5  * Este programa realiza a gestão básica de estoque, verificando se a 
    6  * quantidade atual atende ao mínimo necessário. Caso falte produto, 
    7  * o sistema calcula a reposição necessária e o custo total.
    8  * 
    9  * @author SENAI - Cristiano Batista Pessoa
   10  * @date 15/04/2026
   11  */
   12 
   13 #include <iostream> // Permite a entrada e saída de dados (cin, cout, endl)
   14 #include <iomanip>  // Necessário para formatação de moeda e alinhamento
   15 
   16 using namespace std; // Define o uso do escopo padrão para evitar o prefixo 
   17                      // 'std::' antes de comandos como cout e cin.
   18 
   19 int main()
   20 {
   21     // Configuração para exibir valores monetários com 2 casas decimais
   22     cout << fixed << setprecision(2);
   23 
   24     // Variáveis do sistema
   25     int estoqueAtual, quantidadeMinima, quantidadeReposicao;
   26     double precoUnitario, custoTotal;
   27 
   28     // --- 1. Entrada de Dados ---
   29     cout << "===============================================" << endl;
   30     cout << "       SISTEMA DE ANÁLISE DE ESTOQUE           " << endl;
   31     cout << "===============================================" << endl;
   32 
   33     cout << "Digite a quantidade atual em estoque: ";
   34     cin >> estoqueAtual;
   35 
   36     cout << "Digite a quantidade mínima ideal: ";
   37     cin >> quantidadeMinima;
   38 
   39     cout << "Digite o preço unitário do produto (R$): ";
   40     cin >> precoUnitario;
   41 
   42     // --- 2. Processamento e Verificação ---
   43     cout << "\n          --- RESULTADO DA ANÁLISE ---" << endl;
   44 
   45     if (estoqueAtual < quantidadeMinima)
   46     {
   47         // Cálculo de reposição
   48         quantidadeReposicao = quantidadeMinima - estoqueAtual;
   49         custoTotal = quantidadeReposicao * precoUnitario;
   50 
   51         // --- 3. Exibição (Necessário Repor) ---
   52         cout << "        STATUS: [ REPOSIÇÃO NECESSÁRIA ]" << endl;
   53         cout << "-----------------------------------------------" << endl;
   54         cout << "Quantidade em falta: " << quantidadeReposicao << " unidades." 
   55         << endl;
   56         cout << "Custo da reposição : R$ " << custoTotal << endl;
   57     }
   58     else
   59     {
   60         // --- 3. Exibição (Estoque OK) ---
   61         cout << "         STATUS: [ ESTOQUE SUFICIENTE ]" << endl;
   62         cout << "-----------------------------------------------" << endl;
   63         cout << "Nenhuma ação de compra é necessária no momento." << endl;
   64     }
   65 
   66     cout << "===============================================" << endl;
   67 
   68     // --- 4. Finalização do Programa ---
   69     cout << "\nPressione Enter para finalizar...";
   70     cin.ignore(); // Limpa o buffer do teclado
   71     cin.get();    // Aguarda o usuário pressionar Enter
   72 
   73     return 0;
   74 }
